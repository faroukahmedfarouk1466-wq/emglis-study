مكتبة CustomTkinter — واجهات رسومية حديثة

## ما هي المكتبة؟

CustomTkinter هي مكتبة واجهات رسومية (GUI) مبنية فوق tkinter الأصلية،

بتضيف عليها تصميم حديث، دعم الـ Dark Mode، وتخصيص كامل للألوان والأشكال.

مناسبة جداً لتطبيقات Desktop بواجهة أنيقة.

التثبيت

pip install customtkinter
pip install pillow              # مطلوب لدعم الصور

الاستيراد

import customtkinter as ctk
from PIL import Image           # للصور

إعداد التطبيق

## الإعداد الأولي
ctk.set_appearance_mode("dark")         # "dark" | "light" | "system"
ctk.set_default_color_theme("blue")     # "blue" | "green" | "dark-blue"

إنشاء النافذة الرئيسية

app = ctk.CTk()

app.title("اسم التطبيق")
app.geometry("900x600")                 # العرض × الارتفاع
app.minsize(400, 300)                   # أصغر حجم ممكن
app.maxsize(1920, 1080)                 # أكبر حجم ممكن
app.resizable(True, True)              # السماح بتغيير الحجم
app.configure(fg_color="#1a1a2e")      # لون خلفية النافذة
app.iconbitmap("icon.ico")             # أيقونة النافذة (Windows)
app.attributes("-topmost", True)       # تثبيت فوق النوافذ الأخرى
app.attributes("-fullscreen", True)    # وضع ملء الشاشة

تشغيل التطبيق

app.mainloop()                          # لازم يكون آخر سطر

دالة عند إغلاق النافذة

def on_close():
    print("التطبيق بيتقفل")
    app.destroy()
app.protocol("WM_DELETE_WINDOW", on_close)

CTkEntry — حقل الإدخال

entry = ctk.CTkEntry(

    master=app,
    placeholder_text="اكتب هنا...",
    placeholder_text_color="#555555",   # لون النص الوهمي
    font=ctk.CTkFont("Consolas", 13),
    width=240,
    height=36,
    fg_color="#0d1117",                 # لون الخلفية
    border_color="#0077b6",             # لون الحد العادي
    border_width=2,
    text_color="#e0e0e0",               # لون النص
    corner_radius=8,
    show="*",                           # إخفاء النص (للباسورد)
    state="normal",                     # "normal" | "disabled" | "readonly"
    justify="left",                     # محاذاة النص
    exportselection=True,

)

entry.pack(padx=10, pady=5)

# القراءة والكتابة

entry.get() # قراءة النص

entry.insert(0, "نص افتراضي") # كتابة نص (0 = من البداية)

entry.insert("end", " إضافة") # إضافة في النهاية

entry.delete(0, "end") # حذف كل النص

entry.delete(0, 5) # حذف من 0 لـ 5

# ربط الأحداث

entry.bind("<Return>", lambda e: print(entry.get())) # عند Enter

entry.bind("<FocusIn>", lambda e: print("تم التركيز")) # عند التركيز

entry.bind("<FocusOut>", lambda e: print("خرج التركيز")) # عند المغادرة

entry.bind("<KeyRelease>", lambda e: print(entry.get())) # عند كل ضغطة

CTkToplevel — نافذة منبثقة

def open_popup():
    popup = ctk.CTkToplevel(app)
    popup.title("نافذة جديدة")
    popup.geometry("400x300")
    popup.resizable(False, False)
    popup.grab_set()                   # تحويلها لـ modal (تمنع التفاعل مع النافذة الأم)
    popup.focus()                      # التركيز عليها
    ctk.CTkLabel(popup, text="محتوى النافذة المنبثقة").pack(pady=20)
    ctk.CTkButton(popup, text="إغلاق", command=popup.destroy).pack()
ctk.CTkButton(app, text="فتح نافذة", command=open_popup).pack()

إدارة التخطيط (Layout)

## pack — الأبسط والأكثر استخداماً

widget.pack(

    side="top",            # "top" | "bottom" | "left" | "right"
    fill="both",           # "none" | "x" | "y" | "both"
    expand=True,           # هل يتوسع لملء المساحة؟
    padx=10,               # مسافة أفقية خارجية
    pady=5,                # مسافة رأسية خارجية
    ipadx=5,               # مسافة أفقية داخلية
    ipady=5,               # مسافة رأسية داخلية
    anchor="center",       # "n"|"s"|"e"|"w"|"ne"|"nw"|"se"|"sw"|"center"
    before=other_widget,   # إضافة قبل widget معين
    after=other_widget,    # إضافة بعد widget معين

)

أمثلة عملية

## مثال: نافذة تسجيل دخول

app = ctk.CTk()

app.title("تسجيل الدخول")
app.geometry("400x300")
ctk.set_appearance_mode("dark")

frame = ctk.CTkFrame(app, corner_radius=15)

frame.pack(pady=40, padx=40, fill="both", expand=True)

ctk.CTkLabel(frame, text="تسجيل الدخول", font=ctk.CTkFont(size=20, weight="bold")).pack(pady=20)

username = ctk.CTkEntry(frame, placeholder_text="اسم المستخدم", width=300)

username.pack(pady=10)

password = ctk.CTkEntry(frame, placeholder_text="كلمة المرور", show="*", width=300)

password.pack(pady=10)

def login():
    u = username.get()
    p = password.get()
    print(f"المستخدم: {u}, الباسورد: {p}")
ctk.CTkButton(frame, text="دخول", command=login, width=300).pack(pady=20)
app.mainloop()
